<task type="auto">
  <name>Task 1: Scaffold Vite + React + TypeScript project</name>
  <files>package.json, tsconfig.json, tsconfig.app.json, tsconfig.node.json, vite.config.ts, index.html, src/main.tsx, src/App.tsx</files>
  <action>
    Create the project using Vite's official template:
    ```bash
    npm create vite@latest . -- --template react-ts
    ```
    If the directory is not empty, use a temp directory and move files:
    ```bash
    cd /tmp && npm create vite@latest yt-scaffold -- --template react-ts
    ```
    Then copy scaffold files into the project root (do NOT overwrite .planning/).

    After scaffold is in place:
    1. Run `npm install`
    2. Configure TypeScript strict mode: ensure `"strict": true` in tsconfig.app.json
    3. Add path alias in tsconfig.app.json:
       ```json
       "baseUrl": ".",
       "paths": { "@/*": ["./src/*"] }
       ```
    4. Install @types/node: `npm install -D @types/node`
    5. Configure path alias in vite.config.ts:
       ```typescript
       import path from 'path'
       import react from '@vitejs/plugin-react'
       import { defineConfig } from 'vite'

       export default defineConfig({
         plugins: [react()],
         resolve: {
           alias: {
             '@': path.resolve(__dirname, './src'),
           },
         },
       })
       ```
    6. Delete default Vite boilerplate CSS (src/App.css, src/index.css) — we will use globals.css with Tailwind instead.
    7. Update src/App.tsx to a minimal placeholder:
       ```tsx
       function App() {
         return (
           <div className="min-h-screen flex items-center justify-center">
             <h1 className="text-2xl font-bold">YouTube Knowledge Extractor</h1>
           </div>
         )
       }
       export default App
       ```
    8. Verify `npm run dev` starts without errors.
  </action>
  <verify>
    Run `npm run dev` — Vite dev server starts on localhost.
    Run `npx tsc --noEmit` — TypeScript compiles without errors.
  </verify>
  <done>
    Vite dev server runs, TypeScript strict mode enabled, path aliases configured in both tsconfig and vite.config.
  </done>
</task>

<task type="auto">
  <name>Task 2: Install Tailwind CSS v4 and shadcn/ui</name>
  <files>package.json, src/styles/globals.css, src/lib/utils.ts, components.json, postcss.config.js, src/main.tsx</files>
  <action>
    1. Install Tailwind CSS v4 and PostCSS plugin:
       ```bash
       npm install tailwindcss @tailwindcss/postcss postcss
       ```

    2. Create postcss.config.js (Tailwind v4 uses PostCSS plugin, not tailwind.config.js):
       ```js
       export default {
         plugins: {
           '@tailwindcss/postcss': {}
         }
       }
       ```

    3. Install shadcn/ui dependencies:
       ```bash
       npm install class-variance-authority clsx tailwind-merge lucide-react tw-animate-css
       ```

    4. Create src/lib/utils.ts with the cn() utility:
       ```typescript
       import { type ClassValue, clsx } from "clsx"
       import { twMerge } from "tailwind-merge"

       export function cn(...inputs: ClassValue[]) {
         return twMerge(clsx(inputs))
       }
       ```

    5. Create src/styles/globals.css with Tailwind v4 CSS-first configuration:
       ```css
       @import "tailwindcss";
       @import "tw-animate-css";

       @custom-variant dark (&:is(.dark *));

       @theme inline {
         --color-background: oklch(100% 0 0);
         --color-foreground: oklch(10% 0 0);
         --color-card: oklch(100% 0 0);
         --color-card-foreground: oklch(10% 0 0);
         --color-primary: oklch(15% 0 0);
         --color-primary-foreground: oklch(98% 0 0);
         --color-secondary: oklch(96% 0 0);
         --color-secondary-foreground: oklch(15% 0 0);
         --color-muted: oklch(96% 0 0);
         --color-muted-foreground: oklch(45% 0 0);
         --color-accent: oklch(96% 0 0);
         --color-accent-foreground: oklch(15% 0 0);
         --color-destructive: oklch(55% 0.2 25);
         --color-destructive-foreground: oklch(98% 0 0);
         --color-border: oklch(90% 0 0);
         --color-input: oklch(90% 0 0);
         --color-ring: oklch(15% 0 0);
         --radius-sm: 0.25rem;
         --radius-md: 0.375rem;
         --radius-lg: 0.5rem;
         --radius-xl: 0.75rem;
       }

       @layer base {
         * {
           @apply border-border;
         }
         body {
           @apply bg-background text-foreground;
         }
       }
       ```

    6. Update src/main.tsx to import globals.css:
       ```tsx
       import React from 'react'
       import ReactDOM from 'react-dom/client'
       import App from './App.tsx'
       import './styles/globals.css'

       ReactDOM.createRoot(document.getElementById('root')!).render(
         <React.StrictMode>
           <App />
         </React.StrictMode>,
       )
       ```

    7. Run `npx shadcn@latest init` interactively or create components.json manually:
       ```json
       {
         "$schema": "https://ui.shadcn.com/schema.json",
         "style": "default",
         "rsc": false,
         "tsx": true,
         "tailwind": {
           "config": "",
           "css": "src/styles/globals.css",
           "baseColor": "neutral",
           "cssVariables": true
         },
         "aliases": {
           "components": "@/components",
           "utils": "@/lib/utils",
           "ui": "@/components/ui",
           "lib": "@/lib",
           "hooks": "@/hooks"
         },
         "iconLibrary": "lucide"
       }
       ```

    8. Install initial shadcn/ui components needed for Phase 1:
       ```bash
       npx shadcn@latest add button input label card
       ```

    9. Verify by updating App.tsx to import and render a Button:
       ```tsx
       import { Button } from '@/components/ui/button'

       function App() {
         return (
           <div className="min-h-screen flex items-center justify-center">
             <div className="text-center space-y-4">
               <h1 className="text-2xl font-bold">YouTube Knowledge Extractor</h1>
               <Button>Get Started</Button>
             </div>
           </div>
         )
       }
       export default App
       ```

    10. Run `npm run dev` and verify the Button renders with Tailwind styling.
  </action>
